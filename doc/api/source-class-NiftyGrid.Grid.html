512:     protected function filterData()
513:     {
514:         try{
515:             $filters = array();
516:             foreach($this->filter as $name => $value){
517:                 if(!$this->columnExists($name)){
518:                     throw new UnknownColumnException("Neexistující sloupec $name");
519: 
520:                 }
521:                 if(!$this['columns-'.$name]->hasFilter()){
522:                     throw new UnknownFilterException("Neexistující filtr pro sloupec $name");
523:                 }
524: 
525:                 $type = $this['columns-'.$name]->getFilterType();
526:                 $filter = FilterCondition::prepareFilter($value, $type);
527: 
528:                 if(method_exists("\\NiftyGrid\\FilterCondition", $filter["condition"])){
529:                     $filter = call_user_func("\\NiftyGrid\\FilterCondition::".$filter["condition"], $filter["value"]);
530:                     if(!empty($this['gridForm'][$this->name]['filter'][$name])){
531:                         $filter["column"] = $name;
532:                         if(!empty($this['columns-'.$filter["column"]]->tableName)){
533:                             $filter["column"] = $this['columns-'.$filter["column"]]->tableName;
534:                         }
535:                         $filters[] = $filter;
536:                     }else{
537:                         throw new InvalidFilterException("Neplatný filtr");
538:                     }
539:                 }else{
540:                     throw new InvalidFilterException("Neplatný filtr");
541:                 }
542:             }
543:             return $this->dataSource->filterData($filters);
544:         }
545:         catch(UnknownColumnException $e){
546:             $this->flashMessage($e->getMessage(), "grid-error");
547:             $this->redirect("this", array("filter" => NULL));
548:         }
549:         catch(UnknownFilterException $e){
